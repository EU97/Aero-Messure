\begin{frame}{¿Por qué calibrar el túnel de viento?}
	\begin{columns}
		\begin{column}{0.55\textwidth}
			\begin{itemize}
				\item Ensayos aerodinámicos requieren conocer $V_\infty$ con

				\item El variador de frecuencia (VFD) controla el motor,

\begin{frame}{Principio de Bernoulli -- Tubo de Pitot}
	\begin{columns}
		\begin{column}{0.5\textwidth}
			\textbf{Bernoulli (incompresible):}
			\begin{equation*}
				p + \tfrac{1}{2}\rho V^2 = \text{cte}
			\end{equation*}

			\textbf{En el punto de remanso:}
			\begin{equation*}
				p_0 = p_\infty + \tfrac{1}{2}\rho V_\infty^2
			\end{equation*}

			\vspace{0.3em}
			\begin{block}{Velocidad Pitot}
				\begin{equation*}
					\boxed{V_\infty = \sqrt{\frac{2\,\Delta p}{\rho}}}
				\end{equation*}
				$\Delta p = p_0 - p_s$
			\end{block}
		\end{column}
		\begin{column}{0.47\textwidth}

\begin{frame}{Densidad y Compresibilidad}
	\begin{columns}
		\begin{column}{0.5\textwidth}
			\textbf{Densidad (gas ideal):}
			\begin{equation*}
				\rho = \frac{p_\mathrm{atm}}{R_\mathrm{aire}\,T},
				\quad R_\mathrm{aire} = \SI{287}{\J\per\kg\per\K}
			\end{equation*}

			\textbf{Número de Reynolds:}
			\begin{equation*}
				Re = \frac{V\,L}{\nu}, \quad
				\nu \approx \SI{1.5e-5}{\m\squared\per\s}
			\end{equation*}

			\vspace{0.3em}
			\textbf{Compresibilidad:}
			$Ma < 0{,}3 \Rightarrow$ error $< 1\%$\\

\begin{frame}{Regresión y Calidad del Ajuste}
	\begin{columns}
		\begin{column}{0.5\textwidth}
			\textbf{Modelo lineal:}
			\begin{equation*}
				V(f) = m\,f + b
			\end{equation*}

			\textbf{Modelo polinomial (grado~2):}
			\begin{equation*}
				V(f) = a_2\,f^2 + a_1\,f + a_0
			\end{equation*}

			\vspace{0.3em}
			\textbf{Coeficiente de determinación:}
			\begin{equation*}
				R^2 = 1 - \frac{\sum_i (V_i - \hat{V}_i)^2}
				{\sum_i (V_i - \bar{V})^2}
			\end{equation*}

		\end{column}
		\begin{column}{0.47\textwidth}
			\begin{alertblock}{Funciones clave}
				\begin{itemize}\small
					\item \texttt{fit\_linear()} $\to (m,b,R^2)$
					\item \texttt{fit\_poly2()} $\to$ coefs.
					\item \texttt{summarize\_frequencies()}\\
					      $\to$ media ponderada + $\sigma_\mathrm{pool}$
				\end{itemize}
			\end{alertblock}
			\vspace{0.3em}
			\begin{exampleblock}{Media ponderada}
				\begin{equation*}
					\bar{V}_\mathrm{pool} = \frac{\sum N_i \bar{V}_i}{\sum N_i}
				\end{equation*}
			\end{exampleblock}
		\end{column}
	\end{columns}
\end{frame}

\begin{frame}{Incertidumbre de Velocidad (GUM)}
	\textbf{Propagación de errores:}
	\begin{equation*}
		\frac{u_V}{V} = \frac{1}{2}\sqrt{
			\left(\frac{u_{\Delta p}}{\Delta p}\right)^2
			+ \left(\frac{u_\rho}{\rho}\right)^2}
	\end{equation*}

	\vspace{0.3em}
	\begin{columns}
		\begin{column}{0.5\textwidth}
			\textbf{Ejemplo: $\Delta p = \SI{500}{\Pa}$,
				$\rho = \SI{1.225}{\kg\per\m\cubed}$}
			\begin{itemize}\small
				\item $u_{\Delta p} = \SI{2}{\Pa}$
				\item $u_\rho/\rho = 0{,}3\%$
			\end{itemize}
			\begin{align*}
				V     & = \sqrt{2\times500/1{,}225}
				\approx \SI{28.57}{\m\per\s}        \\
				u_V/V & \approx 0{,}25\%
			\end{align*}
		\end{column}
		\begin{column}{0.47\textwidth}
			\begin{alertblock}{Resultado}
				\centering
				$V = (28{,}57 \pm 0{,}14)\;\si{\m\per\s}$\\
				(\SI{95}{\percent} confianza, $k=2$)
			\end{alertblock}
			\vspace{0.3em}
			\begin{exampleblock}{Incertidumbre LDA}
				\begin{equation*}
					u_{\bar{V}} = \sigma / \sqrt{N}
				\end{equation*}
				Con $N > 1000$: contribución menor
			\end{exampleblock}
		\end{column}
	\end{columns}
\end{frame}
