\begin{frame}{¿Qué es la Anemometría de Hilo Caliente?}
	\begin{columns}
		\begin{column}{0.55\textwidth}
			\begin{itemize}
				\item Sensor: hilo metálico calentado
				      ($d \approx \SI{5}{\micro\metre}$, tungsteno)
				\item Al aumentar $U$ del fluido, el hilo se enfría más
				\item El CTA mantiene $T_w$ cte.\ y mide $E(t)$
				\item \textbf{Resolución temporal} excepcional
				      ($> \SI{100}{\kilo\hertz}$)
			\end{itemize}
			\vspace{0.3em}
			\begin{alertblock}{Objetivo}
				Calibrar el sensor con la \textbf{ley de King} y
				caracterizar la turbulencia del túnel

\begin{frame}{Aplicaciones y Limitaciones}
	\begin{columns}
		\begin{column}{0.5\textwidth}
			\begin{enumerate}\small
				\item Turbulencia en secciones de prueba
				\item Capas límite sobre perfiles aerodinámicos
				\item Estelas y mezcla en toberas
				\item Validación de CFD en régimen turbulento
				\item Verificación de la ley de Kolmogorov
				      ($f^{-5/3}$ en la PSD)
			\end{enumerate}
			\vspace{0.3em}
			\begin{exampleblock}{Ventaja clave}
				Ninguna otra técnica iguala su resolución
				temporal para medir turbulencia
			\end{exampleblock}
		\end{column}
		\begin{column}{0.47\textwidth}
			\begin{alertblock}{Limitaciones}
				\begin{itemize}\small
					\item Sensor frágil ($d \approx \SI{5}{\micro\metre}$)
					\item Susceptible a contaminación
					\item Intrusivo (perturba el flujo)
					\item Relación $E$--$U$ \textbf{no lineal}
					      $\to$ calibración obligatoria
				\end{itemize}
			\end{alertblock}
		\end{column}
	\end{columns}

\begin{frame}{Transferencia de Calor en el Hilo Sensor}
	\begin{columns}
		\begin{column}{0.5\textwidth}
			\textbf{Balance energético:}
			\begin{equation*}
				Q = I^2 R_w = \pi d\,l\,k\,\frac{Nu}{d}\,(T_w - T_f)
			\end{equation*}

			\textbf{Correlación de Kramers:}
			\begin{equation*}
				Nu = 0{,}42\,Pr^{0{,}2}
				+ 0{,}57\,Pr^{0{,}33}\,Re^{0{,}50}
			\end{equation*}

			\vspace{0.3em}
			\textbf{Relación de King (1914):}
			\begin{equation*}
				Q = (T_w - T_f)(k_0 + k_1\sqrt{U})
			\end{equation*}

\begin{frame}{Ley de King -- Forma Clásica y Linealizada}
	\begin{columns}
		\begin{column}{0.5\textwidth}
			\textbf{Forma clásica:}
			\begin{equation*}
				\boxed{E^2 = A' + B'\,U^n}
			\end{equation*}
			$n \approx 0{,}45$--$0{,}50$ (ideal)

			\vspace{0.5em}
			\textbf{Forma linealizada ($n = 3$):}
			\begin{equation*}
				\boxed{U^{1/n} = A\,E^2 + B}
			\end{equation*}

			Variables transformadas:
			\begin{itemize}\small
				\item $X = E^2$ \quad [V$^2$]
				\item $Y = U^{1/3}$ \quad [(mm/s)$^{1/3}$]
			\end{itemize}
			$\Rightarrow$ Regresión lineal $Y = AX + B$
		\end{column}
		\begin{column}{0.47\textwidth}
			\textbf{Reconstrucción de velocidad:}
			\begin{equation*}
				\boxed{U = (A\,E^2 + B)^n}
			\end{equation*}

			\vspace{0.5em}
			\begin{tikzpicture}[scale=0.65]
				\draw[->] (0,0) -- (5.5,0)
				node[right,font=\tiny] {$E^2$ [V$^2$]};
				\draw[->] (0,0) -- (0,4.5)
				node[above,font=\tiny] {$U^{1/3}$};
				% Linear fit
				\draw[aerogreen,thick] (0.5,1.5) -- (5.0,4.0);
				\node[aerogreen,font=\tiny,rotate=27] at (3.5,3.2)
				{$Y = AX + B$};
				% Data points
				\foreach \x/\y in {0.8/1.7, 1.5/2.1, 2.2/2.5,
						3.0/3.0, 3.8/3.4, 4.5/3.8} {

					}
				\node[font=\tiny] at (1.0,0.5) {$R^2 > 0{,}99$};
			\end{tikzpicture}
		\end{column}
	\end{columns}
\end{frame}

\begin{frame}{Estadísticos de Turbulencia}
	\begin{columns}
		\begin{column}{0.5\textwidth}
			\textbf{Velocidad media:}
			\begin{equation*}
				\bar{U} = \frac{1}{N}\sum_{i=1}^{N} U_i
			\end{equation*}

			\textbf{Fluctuación RMS:}
			\begin{equation*}
				u'_\mathrm{rms} = \sqrt{\frac{1}{N-1}
					\sum_{i=1}^{N}(U_i - \bar{U})^2}
			\end{equation*}

			\textbf{Intensidad de turbulencia:}
			\begin{equation*}
				\boxed{I_u = \frac{u'_\mathrm{rms}}{\bar{U}}
					\times 100\%}
			\end{equation*}
		\end{column}
		\begin{column}{0.47\textwidth}
			\begin{tikzpicture}[scale=0.65]
				\draw[->] (0,0) -- (5.5,0)
				node[right,font=\tiny] {$t$};

\begin{frame}{Densidad Espectral de Potencia (PSD) -- Kolmogorov}
	\begin{columns}
		\begin{column}{0.47\textwidth}
			\textbf{Ley de Kolmogorov:}
			\begin{equation*}
				\boxed{S(f) = C_K\,\varepsilon^{2/3}\,f^{-5/3}}
			\end{equation*}
			$C_K \approx 0{,}5$, $\varepsilon$ = tasa de disipación

			\vspace{0.3em}
			\begin{itemize}\small
				\item En log-log: pendiente $-5/3$
				\item Confirma turbulencia desarrollada
				\item Tres zonas: energía, inercial, disipación
			\end{itemize}
		\end{column}
		\begin{column}{0.5\textwidth}
			\begin{tikzpicture}[scale=0.65]

\begin{frame}{Consolidación y Transformación}
	\begin{columns}
		\begin{column}{0.5\textwidth}
			\textbf{Promedio por iteración (Paso 2):}
			\begin{equation*}
				E_\mathrm{iter} = \frac{1}{N_\mathrm{útil}}
				\sum_{i=1}^{N_\mathrm{útil}} E_i
			\end{equation*}

			\textbf{Consolidación por SD (Paso 3):}
			\begin{equation*}
				E_\mathrm{SD} = \frac{1}{N_\mathrm{iter}}
				\sum_{j=1}^{N_\mathrm{iter}} E_{\mathrm{iter},j}
			\end{equation*}

			\textbf{Transformación (Paso 4):}
			\begin{align*}
				X & = E_\mathrm{SD}^2                  \\
				Y & = U_\mathrm{ref}^{1/n} \quad (n=3)
			\end{align*}
		\end{column}
		\begin{column}{0.47\textwidth}
			\begin{exampleblock}{Ejemplo: SD~$=25\%$}
				\scriptsize
				\begin{itemize}
					\item 7 iteraciones $\to$ 7 valores
					      $E_\mathrm{iter}$
					\item $E_\mathrm{SD} = 2{,}740$ V
					\item $E^2 = 7{,}508$ V$^2$
					\item $U_\mathrm{ref} = 188{,}7$ mm/s
					\item $U^{1/3} = 5{,}736$
				\end{itemize}
			\end{exampleblock}
			\vspace{0.3em}
			\begin{block}{Repetibilidad}
				$\sigma_{E,\mathrm{SD}}$ cuantifica la
				variabilidad entre iteraciones\\
				$\to$ fuente principal de $u_E$
			\end{block}
		\end{column}
	\end{columns}
\end{frame}

\section{Incertidumbre}
% =============================================================================

\begin{frame}{Propagación GUM -- Ley de King}
	\textbf{Incertidumbre combinada:}
	\begin{equation*}
		u_U^2 = \left(\frac{\partial U}{\partial E}\right)^2 u_E^2
		+ \left(\frac{\partial U}{\partial A}\right)^2 u_A^2
		+ \left(\frac{\partial U}{\partial B}\right)^2 u_B^2
	\end{equation*}

	\vspace{0.3em}
	\begin{columns}
		\begin{column}{0.5\textwidth}
			\textbf{Sensibilidades:}
			\begin{align*}
				\frac{\partial U}{\partial E}
				 & = 2nAE(AE^2+B)^{n-1} \\
				\frac{\partial U}{\partial A}
				 & = nE^2(AE^2+B)^{n-1} \\
				\frac{\partial U}{\partial B}
				 & = n(AE^2+B)^{n-1}
			\end{align*}
		\end{column}
		\begin{column}{0.47\textwidth}
			\begin{exampleblock}{Valores típicos}
				\begin{itemize}\small
					\item $E \sim \SI{2.5}{\volt}$,
					      $u_E \sim \SI{0.02}{\volt}$
					\item Contribución dominante: $u_E$
					\item Incertidumbre relativa:
					      $u_U/U \sim 2$--$5\%$
				\end{itemize}
			\end{exampleblock}
			\vspace{0.2em}
			\begin{block}{Función del script}
				\texttt{propagate\_sigma\_U(df, A, B, n)}\\
				$\to$ columna $u_U$ por cada SD
			\end{block}
		\end{column}
	\end{columns}
\end{frame}

\section{Conclusiones}
% =============================================================================

\begin{frame}{Conclusiones}
	\begin{enumerate}
		\item La ley de King con $n=3$ linealizada como
		      $U^{1/3} = 0{,}274\,E^2 + 3{,}475$ describe la
		      relación $E$--$U$ ($R^2 = 0{,}984$)
		\item El procesamiento en 4 etapas (descarte, promedio,
		      consolidación, transformación) proporciona puntos de
		      calibración robustos
		\item Intensidad de turbulencia $I_u \approx 5$--$9\%$
		      clasifica el flujo como \textbf{moderadamente turbulento}
		\item La PSD confirma la ley $f^{-5/3}$ de Kolmogorov en
		      el sub-rango inercial
		\item Contaminación y $\Delta T_f$ son las principales
		      causas de deriva de calibración
		\item Riesgo principal: \textbf{rotura del hilo}
		      ($R = 16$, Alto) -- mitigable con cuidado extremo
		\item \texttt{P03\_Hilo\_Caliente.py} automatiza todo el
		      flujo desde datos crudos hasta la curva de calibración
	\end{enumerate}
