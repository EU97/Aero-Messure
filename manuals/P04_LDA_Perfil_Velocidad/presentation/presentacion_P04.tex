\section{Principio LDA}
\begin{frame}{Efecto Doppler y Modelo de Franjas}
  \begin{columns}
    \begin{column}{0.5\textwidth}
      \textbf{Frecuencia Doppler:}
      \begin{equation*}
        f_D = \frac{2\sin(\theta/2)}{\lambda}\,U
      \end{equation*}
      \textbf{Separación de franjas:}
      \begin{equation*}
        \delta_f = \frac{\lambda}{2\sin(\theta/2)}
      \end{equation*}
      \begin{block}{Velocidad}
        \begin{equation*}
          \boxed{U = f_D \cdot \delta_f}
        \end{equation*}
      \end{block}
      \textbf{Ventajas vs HWA:}
      \begin{itemize}\small
        \item No intrusiva (sin sonda en el flujo)
        \item Mide flujos inversos (celda de Bragg)
        \item No requiere calibración
      \end{itemize}
    \end{column}

\section{Cálculo Numérico}
\begin{frame}{Parámetros del sistema LDA}
  \begin{columns}
    \begin{column}{0.5\textwidth}
      \textbf{Sistema Ar-Ion del laboratorio:}
      \begin{itemize}
        \item $\lambda = \SI{514.5}{\nm}$
        \item $\theta/2 = \SI{4.5}{\degree}$
        \item $f_L = \SI{250}{\mm}$, $d_b = \SI{1.2}{\mm}$
      \end{itemize}
      \vspace{0.5em}
      \textbf{Separación de franjas:}
      \begin{align*}
        \delta_f &= \frac{514.5\times10^{-9}}{2\sin(4.5°)}\\
                 &= \SI{3.279}{\micro\metre}
      \end{align*}
      \textbf{$U$ para $f_D = \SI{1.53}{\MHz}$:}
      \begin{equation*}
        U = 1.53\!\times\!10^6 \times 3.279\!\times\!10^{-6}
          \approx \SI{5.0}{\m\per\s}
      \end{equation*}
    \end{column}
    \begin{column}{0.47\textwidth}
      \textbf{Volumen de medida:}
      \begin{align*}
        d_{mv} &= \frac{4 f_L \lambda}{\pi d_b}\\
               &\approx \SI{136.5}{\micro\m}\\
        l_{mv} &= \frac{d_{mv}}{\sin(\theta/2)} \approx \SI{1.74}{\mm}\\
        N_f    &= d_{mv}/\delta_f \approx 42\text{ franjas}
      \end{align*}
      \begin{alertblock}{}
        \centering
        $u_U \approx \SI{0.033}{\m\per\s}$\\($\approx 0.16\%$ a \SI{20}{\m\per\s})
      \end{alertblock}
    \end{column}
  \end{columns}
\end{frame}

\section{Caso: Estela NACA 0012}
\begin{frame}{Coeficiente de Arrastre por Estela}
  \begin{block}{Método de Jones}
    \begin{equation*}
      C_D = \frac{2}{c}\int_{-\infty}^{+\infty}
            \frac{U_1}{U_\infty}\!\left(1 - \frac{U_1}{U_\infty}\right)dy
      \approx 0.012
    \end{equation*}
    NACA 0012, $Re_c=1.5\times10^5$, $\alpha=0°$ (consistente con literatura)
  \end{block}
  \begin{columns}
    \begin{column}{0.5\textwidth}
      \begin{itemize}
        \item $c = \SI{150}{\mm}$, medida a $x/c=1.0$
        \item Barrido $y \in [-50,50]\,\si{\mm}$, $\Delta y = \SI{5}{\mm}$
        \item Mínimo $U_1/U_\infty \approx 0.72$ en la línea central
        \item Integración trapezoidal con datos LDA
      \end{itemize}
    \end{column}

import numpy as np
from pathlib import Path

def read_lda(folder):
    data = []
    for f in sorted(folder.glob("*.txt")):
        d = np.loadtxt(f)
        data.append(d[:,1])  # U
    return np.concatenate(data)

def filter_iqr(U, k=1.5):
    Q1, Q3 = np.percentile(U, [25,75])
    IQR = Q3 - Q1
    mask = (U >= Q1-k*IQR) & (U <= Q3+k*IQR)
    return U[mask]

root = Path("files/P4")
for folder in sorted(root.iterdir()):
    U = filter_iqr(read_lda(folder))
    print(f"{folder.name}: "
          f"U={U.mean():.2f} m/s, "
          f"Iu={U.std()/U.mean()*100:.1f}%")
