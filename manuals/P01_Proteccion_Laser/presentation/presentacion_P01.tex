\begin{frame}{Exposición Máxima Permisible (MPE)}
  \begin{columns}
    \begin{column}{0.5\textwidth}
      La MPE es el nivel de irradiancia u exposición radiante al que puede
      estar expuesto el ojo \textbf{sin daño}.

      \vspace{0.5em}
      Para $\lambda = \SI{532}{\nm}$, $t = \SI{6}{\ns}$:
      \begin{equation*}
        \mathrm{MPE} = \SI{5e-3}{\joule\per\metre\squared}
      \end{equation*}
      (IEC 60825-1:2014, Tabla A.1)

      \vspace{0.5em}
      Depende de:
      \begin{itemize}
        \item Longitud de onda $\lambda$
        \item Duración exposición $t$
        \item Ángulo de subtensa $\alpha$
      \end{itemize}
    \end{column}

\begin{frame}{Densidad Óptica (OD)}
  \begin{block}{Definición}
    \begin{equation*}
      \boxed{OD = \log_{10}\!\left(\frac{H_0}{H_{\mathrm{MPE}}}\right)}
    \end{equation*}
    \begin{itemize}
      \item $H_0$ = exposición incidente [\si{\joule\per\metre\squared}]
      \item $H_{\mathrm{MPE}}$ = MPE para $\lambda$ y $t$
    \end{itemize}
  \end{block}

  \vspace{0.3em}
  \textbf{Exposición del haz gaussiano:}
  \begin{equation*}
    H_0 = \frac{Q}{\pi w_0^2}
  \end{equation*}

  \vspace{0.3em}
  \textbf{Transmitancia del filtro:} $T = 10^{-OD}$
  \begin{equation*}
    H_{\mathrm{trans}} = H_0 \cdot 10^{-OD} \leq H_{\mathrm{MPE}}
    \quad \Rightarrow \quad
    OD \geq \log_{10}\!\left(\frac{H_0}{H_{\mathrm{MPE}}}\right)
  \end{equation*}
\end{frame}

\section{Cálculo Numérico}

\begin{frame}{Ejemplo: PIV con Nd:YAG 532 nm}
  \begin{columns}
    \begin{column}{0.45\textwidth}
      \textbf{Datos del sistema:}
      \begin{itemize}
        \item $Q = \SI{200}{\milli\joule}$
        \item $\tau = \SI{6}{\ns}$
        \item $w_0 = \SI{5}{\mm}$
        \item $\phi = \SI{3}{\milli\radian}$
        \item $f_\mathrm{rep} = \SI{10}{\Hz}$
      \end{itemize}
    \end{column}
    \begin{column}{0.52\textwidth}
      \textbf{Paso 1 -- Exposición incidente:}
      \begin{align*}
        H_0 &= \frac{0.200}{\pi (5\times10^{-3})^2}\\
            &\approx \SI{2547}{\joule\per\metre\squared}
      \end{align*}

      \textbf{Paso 2 -- MPE:} $\SI{5e-3}{\joule\per\metre\squared}$

      \textbf{Paso 3 -- OD:}
      \begin{align*}
        OD &= \log_{10}\!\left(\frac{2547}{5\times10^{-3}}\right)\\
           &= \log_{10}(5.09\times10^5) \approx 5.71
      \end{align*}
      \begin{alertblock}{}
        \centering \textbf{OD requerido = 6}\\
        Marcado: \texttt{I 532 L6}
      \end{alertblock}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Zona Nominal de Peligro (NHZ)}
  \begin{columns}
    \begin{column}{0.5\textwidth}
      \begin{equation*}
        r_{\mathrm{NHZ}} = \frac{1}{\phi}
          \sqrt{\frac{4P_\mathrm{pico}}{\pi E_\mathrm{MPE}}}
      \end{equation*}

      \vspace{0.3em}
      $P_\mathrm{pico} = Q/\tau = 0.2/6\!\times\!10^{-9}
       = \SI{3.33e7}{\watt}$

      \begin{align*}
        r_{\mathrm{NHZ}} &= \frac{1}{3\!\times\!10^{-3}}
          \sqrt{\frac{4\!\times\!3.33\!\times\!10^7}
                     {\pi\!\times\!8.33\!\times\!10^5}}\\
        &\approx \textbf{\SI{2376}{\metre}}
      \end{align*}
    \end{column}

\section{Caso de Estudio}

\begin{frame}{Caso: PIV sobre NACA 0012 en túnel de viento}
  \begin{block}{Escenario}
    Medición PIV para estudiar transición laminar-turbulenta sobre ala NACA~0012
    a $\mathrm{Re} = 3\times10^5$ en túnel $\SI{0.5}{\m}\times\SI{0.5}{\m}$.
  \end{block}

  \begin{columns}
    \begin{column}{0.5\textwidth}
      \textbf{Sistema PIV:}
      \begin{itemize}\small
        \item Nd:YAG 532 nm, 200 mJ/pulso
        \item $\tau = \SI{6}{\ns}$, $f = \SI{10}{\Hz}$
        \item Lente cilíndrica $f = -\SI{25}{\mm}$
        \item CCD 2 Mpx, $\Delta t = \SI{50}{\micro\second}$
        \item Seeding: aceite nebulizado \SI{1}{\micro\metre}
      \end{itemize}
    \end{column}
    \begin{column}{0.47\textwidth}
      \textbf{Resultados:}
      \begin{itemize}\small
        \item $H_0 = \SI{2547}{\joule\per\metre\squared}$
        \item $\mathrm{MPE} = \SI{5e-3}{\joule\per\metre\squared}$
        \item $OD_{\min} = 5.71 \Rightarrow$ \textbf{OD = 6}
        \item NHZ libre $\approx \SI{2.4}{\km}$
        \item NHZ lámina $\approx \SI{71}{\m}$
      \end{itemize}
      \vspace{0.3em}
      \begin{alertblock}{}
        \centering \texttt{I 532 L6} + Interlock
      \end{alertblock}
    \end{column}
  \end{columns}
\end{frame}

def calcular_OD(Q, w0, MPE):
    H0 = Q / (np.pi * w0**2)
    OD = np.log10(H0 / MPE)
    return OD, H0

# Nd:YAG 532nm PIV

\section{Resumen}

\begin{frame}{Conclusiones}
  \begin{enumerate}
    \item El PIV con Nd:YAG \SI{532}{\nm} es \textbf{Clase 4} -- máximo peligro.
    \item Se requiere \textbf{OD = 6} ($H_0/\mathrm{MPE} \approx 5\times10^5$),
          marcado \texttt{I 532 L6} (EN 207).
    \item La \textbf{NHZ} para el haz libre supera los \SI{2}{\km} --
          controles de ingeniería son obligatorios.
    \item La \textbf{jerarquía de controles}: ingeniería $>$ administrativo $>$ EPP.
    \item El \textbf{notebook Python} automatiza los cálculos y permite adaptar
          la evaluación a cualquier configuración.
  \end{enumerate}
